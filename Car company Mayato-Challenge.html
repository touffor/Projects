makes = []
cols = []
means = []
medians = []
standard_deviations = []
coef_vars = []

for group in df.groupby('Make'):

    make = group[0]
    data = group[1][Columns_neeeded]
    
    for Name in data.columns:

        mean = data[f'{Name}'].mean()
        median = data[f'{Name}'].median()
        std = data[f'{Name}'].std()
        coef_var = data[f'{Name}'].std()/data[f'{Name}'].mean()

        cols.append(Name)
        makes.append(make)
        means.append(mean)
        medians.append(median)
        standard_deviations.append(std)
        coef_vars.append(coef_var)

        print(f"{make} | {Name}")
        print(f"Mean:  {mean}")
        print(f"Median: {median}")
        print(f"standard deviation: {std}")
        print(f"Coefficient of Variance {coef_var}")
